    public String createJwt(User user) {

        claims.put("id", user.getId());
        claims.put("email", user.getEmail());
        return Jwts.builder()

                    .setExpiration(new Date(System.currentTimeMillis() + JWT_EXPIRATION))
                    .signWith(SignatureAlgorithm.HS512, JWT_SECRET)
                    .compact();
    }
